\begin{problem}{A + B}{1s}{512 MiB}{stdin}{stdout}
Cho 2 số nguyên dương $a$ và $b$. Hãy tính tổng $a + b$.

\InputFile
Một dòng chứa 2 số nguyên dương $a$ và $b$.

\OutputFile
In ra $a + b$.

\Constraints
$1 \le a, b \le 10^9$.

\Example
\begin{example}
\exmp{
1 2
}{
3
}%
\end{example}
\end{problem}
